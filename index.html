def analyze_social_media_performance(data_file):
    # Load social media data from CSV
    df = pd.read_csv(data_file)
    
    # Calculate engagement rate
    df['engagement_rate'] = ((df['likes'] + df['comments']) / 
                            df['followers']) * 100
    
    # Analyze best posting times
    performance_by_hour = df.groupby('posting_hour')['engagement_rate'].mean()
    
    # Find best performing content types
    content_performance = df.groupby('content_type')['engagement_rate'].mean()
    
    return {
        'best_posting_time': performance_by_hour.idxmax(),
        'best_content_type': content_performance.idxmax(),
        'avg_engagement': df['engagement_rate'].mean()
    }
